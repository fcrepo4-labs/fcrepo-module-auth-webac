50  /**
51   * Insert WebAC Link headers to responses
52   *
53   * @author whikloj
54   * @since 2015-10-30
55   */
56  @Component
57  public class LinkHeaderProvider implements UriAwareHttpHeaderFactory {
58  
59      private static final Logger LOGGER = getLogger(LinkHeaderProvider.class);
60  
61      @Inject
62      private SessionFactory sessionFactory;
63  
64      @Inject
65      private NodeService nodeService;
66  
67      @Override
68      public Multimap<String, String> createHttpHeadersForResource(final UriInfo uriInfo, final FedoraResource resource) {
69  
70          final FedoraSession internalSession = sessionFactory.getInternalSession();
71          final IdentifierConverter<Resource, FedoraResource> translator =
72                  new DefaultIdentifierTranslator(getJcrSession(internalSession));
73          final ListMultimap<String, String> headers = ArrayListMultimap.create();
74  
75          LOGGER.debug("Adding WebAC Link Header for Resource: {}", resource.getPath());
76          // Get the correct Acl for this resource
77          WebACRolesProvider.getEffectiveAcl(resource).ifPresent(acls -> {
78              // If the Acl is present we need to use the internal session to get its URI
79              nodeService.find(internalSession, acls.resource.getPath())
80              .getTriples(translator, PROPERTIES)
81              .collect(toModel()).listObjectsOfProperty(createProperty(WEBAC_ACCESS_CONTROL_VALUE))
82              .forEachRemaining(linkObj -> {
83                  if (linkObj.isURIResource()) {
84                      final Resource acl = linkObj.asResource();
85                      final String aclPath = translator.convert(acl).getPath();
86                      final URI aclUri = uriInfo.getBaseUriBuilder().path(aclPath).build();
87                      headers.put("Link", Link.fromUri(aclUri).rel("acl").build().toString());
88                  }
89              });
90          });
91  
92          return headers;
93      }
94  
95  
96  }
